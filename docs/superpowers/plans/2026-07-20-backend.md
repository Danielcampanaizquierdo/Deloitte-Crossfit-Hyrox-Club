# Backend Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a full server-side backend to the Deno Fresh club app: HMAC session auth for admin, SSR data loading, and API routes for all public and admin form submissions.

**Architecture:** A `_middleware.ts` reads the admin session cookie on every request and writes `ctx.state.isAdmin`. `routes/index.tsx` gains a `Handlers<PageData>` that fetches all approved (+ pending for admin) data in parallel and passes it as props. Form modals become hydrated islands that `fetch()` to dedicated `/api/*` routes. All routes return JSON; admin-only routes enforce `ctx.state.isAdmin`.

**Tech Stack:** Deno Fresh 1.6.8, Preact 10.19.3, `crypto.subtle` (Web Crypto, built into Deno Deploy), JSON file storage via existing `lib/storage.ts`.

## Global Constraints

- Deno Fresh 1.6.8 — imports use `"$fresh/server.ts"` for `Handlers`, `PageProps`, `MiddlewareHandlerContext`
- Standard library prefix is `"std/"` (already in `deno.json` import map pointing to `https://deno.land/std@0.208.0/`)
- All API routes return `Response.json(body, { status })` — no hand-rolled `new Response(JSON.stringify(...), { headers: {...} })`
- Env vars: `ADMIN_PASSCODE` and `SESSION_SECRET` — never hardcode secrets in source
- Cookie name: `admin_session`
- Session TTL: 28800 seconds (8 hours)
- All new files in `routes/api/` are handler-only (no default export)
- `fresh.gen.ts` is auto-generated — never edit it manually; run `deno task dev` after adding islands

- [ ] **Step 1: Create `lib/session.ts`**

```ts
const COOKIE_NAME = "admin_session";

function secret(): string {
  const s = Deno.env.get("SESSION_SECRET");
  if (!s) throw new Error("SESSION_SECRET env var not set");
  return s;
}

async function sign(token: string): Promise<string> {
  const key = await crypto.subtle.importKey(
    "raw",
    new TextEncoder().encode(secret()),
    { name: "HMAC", hash: "SHA-256" },
    false,
    ["sign"],
  );
  const sig = await crypto.subtle.sign("HMAC", key, new TextEncoder().encode(token));
  return btoa(String.fromCharCode(...new Uint8Array(sig)));
}

export async function createSession(): Promise<string> {
  const token = btoa(
    String.fromCharCode(...crypto.getRandomValues(new Uint8Array(32))),
  );
  const sig = await sign(token);
  const value = encodeURIComponent(`${token}.${sig}`);
  const secure = Deno.env.get("DENO_DEPLOYMENT_ID") ? "; Secure" : "";
  return `${COOKIE_NAME}=${value}; HttpOnly; SameSite=Lax; Path=/; Max-Age=28800${secure}`;
}

export async function verifySession(cookieHeader: string): Promise<boolean> {
  const match = cookieHeader
    .split(";")
    .map((c) => c.trim())
    .find((c) => c.startsWith(`${COOKIE_NAME}=`));
  if (!match) return false;

  const raw = decodeURIComponent(match.slice(COOKIE_NAME.length + 1));
  const dot = raw.lastIndexOf(".");
  if (dot === -1) return false;

  const token = raw.slice(0, dot);
  const sig = raw.slice(dot + 1);

  try {
    const expected = await sign(token);
    if (expected.length !== sig.length) return false;
    let diff = 0;
    for (let i = 0; i < expected.length; i++) {
      diff |= expected.charCodeAt(i) ^ sig.charCodeAt(i);
    }
    return diff === 0;
  } catch {
    return false;
  }
}

export function clearSession(): string {
  const secure = Deno.env.get("DENO_DEPLOYMENT_ID") ? "; Secure" : "";
  return `${COOKIE_NAME}=; HttpOnly; SameSite=Lax; Path=/; Max-Age=0${secure}`;
}
```

import { createSession } from "../../../lib/session.ts";
import { State } from "../../../types/State.ts";

export const handler: Handlers<unknown, State> = {
  async POST(req, _ctx) {
    let body: { passcode?: string };
    try {
      body = await req.json();
    } catch {
      return Response.json({ error: "JSON inválido" }, { status: 400 });
    }

    if (!body.passcode) {
      return Response.json({ error: "Passcode requerido" }, { status: 400 });
    }

    const adminPasscode = Deno.env.get("ADMIN_PASSCODE") ?? "";
    if (!adminPasscode) {
      return Response.json({ error: "ADMIN_PASSCODE not configured" }, { status: 500 });
    }

    if (body.passcode !== adminPasscode) {
      return Response.json({ error: "Passcode incorrecto" }, { status: 401 });
    }

    const cookie = await createSession();
    return Response.json({ success: true }, {
      headers: { "set-cookie": cookie },
    });
  },
};

      body = await req.json();

      return Response.json({ error: "JSON inválido" }, { status: 400 });
    }

    const { name, email, level, goal, location } = body;

      name,
      email,
      level: level as "beginner" | "intermediate" | "advanced",
      goal: goal as "crossfit" | "hyrox" | "general",
      location,
      bio: body.bio,
    });

interface AdminPendingPanelProps {

function useApproveReject(initialIds: string[]) {
  const [ids, setIds] = useState(initialIds);
  const remove = (id: string) => setIds((prev) => prev.filter((i) => i !== id));
  return { ids, remove };
}

export default function AdminPendingPanel({ pending }: AdminPendingPanelProps) {
  const events = useApproveReject(pending.events.map((e) => e.id));
  const members = useApproveReject(pending.members.map((m) => m.id));
  const prs = useApproveReject(pending.prs.map((p) => p.id));
  const results = useApproveReject(pending.results.map((r) => r.id));

  const action = async (url: string, method: string, id: string, remove: (id: string) => void) => {
    const res = await fetch(url, { method });
    if (res.ok) remove(id);
  };

  const pendingEvents = pending.events.filter((e) => events.ids.includes(e.id));
  const pendingMembers = pending.members.filter((m) => members.ids.includes(m.id));
  const pendingPRs = pending.prs.filter((p) => prs.ids.includes(p.id));
  const pendingResults = pending.results.filter((r) => results.ids.includes(r.id));

  return (
    <div class="grid cards2" style="margin-top:16px">
      <div class="pending">
        <h3>Eventos pendientes <span class="badge">{pendingEvents.length} pending</span></h3>

                Aprobar
              </button>
              <button class="btn red" style="flex:1;padding:8px"
                onClick={() => action(`/api/events/${ev.id}/delete`, "DELETE", ev.id, events.remove)}>
                Rechazar
              </button>
            </div>
          </div>
        ))}
      </div>

      <div class="pending">
        <h3>PRs pendientes <span class="badge">{pendingPRs.length} pending</span></h3>
        {pendingPRs.length === 0 && <p class="muted">Sin pendientes.</p>}
        {pendingPRs.map((pr) => (
          <div key={pr.id} class="pending-item">
            <b>{pr.memberName} · {pr.movement} · {pr.weight}kg</b><br />
            <span class="muted">{pr.memberEmail}</span>
            <div class="actions" style="margin-top:8px;gap:8px">
              <button class="btn green" style="flex:1;padding:8px"
                onClick={() => action(`/api/prs/${pr.id}/approve`, "POST", pr.id, prs.remove)}>

              <button class="btn red" style="flex:1;padding:8px"
